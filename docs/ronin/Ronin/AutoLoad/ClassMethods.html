def const_missing(name)
  const = super(name)

  # if the loaded Class is a DataMapper Resource, re-finalize
  if const < DataMapper::Resource
    DataMapper.finalize
  end

  return const
end

def require_const(name)
  const = super(name)

  if (const && const < DataMapper::Resource)
    DataMapper.finalize
  end

  return const
end
