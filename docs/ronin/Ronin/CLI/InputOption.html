def self.included(command)
  command.option :input, short: '-i',
                         value: {
                           type:  String,
                           usage: 'FILE'
                         },
                         desc: 'Optional input file'
end

def input_stream(&block)
  if (path = options[:input])
    File.open(path,&block)
  else
    yield stdin
  end
end
