def parse_type_signature(type)
  unless type =~ /\A[_a-z0][_a-z0-9]*(?:\[\d*\])*/
    print_error "invalid type signature: #{type}"
    exit(-1)
  end

  scanner = StringScanner.new(type)
  type    = scanner.scan(/[_a-z0][_a-z0-9]*/)

  if scanner.getch == '['
    array_dimensions = []

    while scanner.scan('[')
      if (length = scanner.scan(/\d+/))
        array_dimensions << length
      else scanner.scan(']')
        array_dimensions << nil
      end
    end

    return type, array_dimensions
  else
    return type
  end
end

def resolve_type(name)

rescue ArgumentError
  print_error("unknown type #{name.inspect}")
  exit(-1)
end
