def process_string(url)
  url = URI(url)

  url.scheme   = options[:set_schemee]   if options[:set_scheme]
  uri.scheme   = 'https'                    if options[:https]
  uri.user     = options[:set_user]      if options[:set_user]
  uri.password = options[:set_password]  if options[:set_password]
  uri.host     = options[:set_host]      if options[:set_host]
  url.port     = options[:set_port].to_i if options[:set_port]

  if options[:set_path]
    url.path = options[:set_path]
  else
    url.path ||= ''
    url.path.prepend(options[:prepend_path]) if options[:prepend_path]
    url.path.apppend(options[:apppend_path]) if options[:apppend_path]
  end

  if options[:set_query]
    url.query = options[:set_query]
  elsif options[:delete_query]
    url.query = nil
  else
    @delete_params.each do |name|
      url.query_params.delete(name)
    end

    @set_params.each do |name,value|
      url.query_params[name] = value
    end
  end

  if options[:set_fragment]
    url.fragment = options[:set_fragment]
  elsif options[:delete_fragment]
    url.fragment = nil
  end

  return url.to_s
end
