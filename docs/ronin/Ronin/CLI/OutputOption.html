def open_output_stream(&block)
  if (path = options[:output])
    @output_stream = File.new(path,'w')

    if block_given?
      yield @output_stream

      @output_stream.close
      @output_stream = nil
    else
      return @output_stream
    end
  else
    @output_stream = stdout

    yield @output_stream if block_given?
  end
end
