def deploy!
  test!

  print_info "Deploying #{self} ..."

  @deployed = false
  @deploy_blocks.each { |block| block.call() }
  @deployed = true
  @evacuated = false

  print_info "#{self} deployed!"

  yield if block_given?
  return self
end

def evacuate!
  yield if block_given?

  print_info "Evauating #{self} ..."

  @evacuated = false
  @evacuate_blocks.each { |block| block.call() }
  @evacuated = true
  @deployed = false

  print_info "#{self} evacuated."
  return self
end
