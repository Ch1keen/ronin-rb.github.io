def run
  self.params = options
  print_debug "#{self} parameters: #{self.params.inspect}"

  print_info "Building #{self} ..."

  @built = false
  @build_blocks.each { |block| block.call() }
  @built = true

  print_info "#{self} built!"

  yield if block_given?
  return self
end
