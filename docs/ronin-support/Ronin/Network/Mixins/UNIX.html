def unix_accept
  print_info "Listening on #{self.path} ..."

  super(self.path) do |client|
    print_info "Client connected to #{self.path}"

    yield client if block_given?

    print_info "Client disconnecting from #{self.path} ..."
  end

  print_info "Closed #{self.path}"
  return nil
end
