def dns_resolver(nameservers: nil, nameserver: nil)
  if nameserver
    Network::DNS.resolver(nameserver: nameserver)
  elsif nameservers
    Network::DNS.resolver(nameservers: nameservers)
  else
    @dns_resolver ||= Network::DNS.resolver(
      nameservers: dns_nameservers
    )
  end
