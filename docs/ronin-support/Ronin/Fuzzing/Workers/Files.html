def each
  return enum_for unless block_given?

  index = 1

  extname  = File.extname(@path)
  basename = @path.chomp(extname)

  super do |data|
    path = "#{basename}-#{index}#{extname}"

    File.binwrite(path,data)
    yield path

    index += 1
  end
end
