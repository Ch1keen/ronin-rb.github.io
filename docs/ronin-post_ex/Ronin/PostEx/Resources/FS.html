Controls the resources of a File System.

def chdir(path)
  path = join(path)
  old_cwd = @cwd

  @cwd = if @api.respond_to?(:fs_chdir)
           @api.fs_chdir(path)
         else
           path
         end

  if block_given?
    yield @cwd
    chdir(old_cwd)
  end

  return @cwd
end

def join(path)
  if (@cwd && path[0,1] != '/')
    ::File.expand_path(::File.join(@cwd,path))
  else
    path
  end
end
