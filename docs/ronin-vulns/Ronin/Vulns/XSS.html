Overview

Represents a Cross Site Scripting (XSS) vulnerability.

Features

  • Tests a URL with just one HTTP request (per param).
  • Tests which HTML special characters are allowed.
  • Includes random data in the test values.

def determine_context(html,index)
  case html[0,index]
  when IN_TAG                         then :html
  when IN_IN_DOUBLE_QUOTED_ATTR_VALUE then :double_quoted_attr_value
  when IN_IN_SINGLE_QUOTED_ATTR_VALUE then :single_quoted_attr_value
  when IN_IN_UNQUOTED_ATTR_VALUE      then :unquoted_attr_value
  when IN_ATTR_NAME                   then :attr_name
  when IN_ATTR_LIST                   then :attr_list
  when IN_TAG_NAME                    then :tag_name
  else                                     :html
  end
end

def determine_viability(context,allowed_chars)
  required_chars = REQUIRED_CHARS.fetch(context) do
    raise(ArgumentError,"unknown context (#{context.inspect}) must be #{REQUIRED_CHARS.map(&:inspect).join(', ')}")
  end

  allowed_chars.superset?(required_chars)
end

def vulnerable?
  key_value = random_value
  xss       = "'#{key_value} =/>();<\"'#{key_value}"
  regexp    = %r{(?:(')|&#39;)#{key_value}( )?(=)?(?:(')|&#39;)?(/)?(?:(>)|&gt;)?(\()?(\))?(;)?(?:(<)|&lt;)?(?:(")|&quote;)?(?:'|&#39;)?#{key_value}}i

  response     = exploit("#{original_value}#{xss}")
  content_type = response.content_type
  body         = response.body

  if content_type && content_type.include?('text/html')
    if (match = body.match(regexp))
      @context       = determine_context(body[0,match.start])
      @allowed_chars = match[1..].compact.to_set

      return determine_viability(@context,@allowed_chars)
    end
  end
end
