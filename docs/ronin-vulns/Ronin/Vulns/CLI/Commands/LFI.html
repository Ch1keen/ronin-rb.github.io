def scan_kwargs
  kwargs = super()

  kwargs[:os]    = options[:os]    if options[:os]
  kwargs[:depth] = options[:depth] if options[:depth]

  if options[:filter_bypass]
    kwargs[:filter_bypass] = options[:filter_bypass]
  end

  return kwargs
end
