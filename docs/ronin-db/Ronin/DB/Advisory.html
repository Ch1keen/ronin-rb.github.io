def self.parse(id)
  if (advisory = find(id))
    advisory
  else
    prefix, year, number = split(id)

    new(
      id:     id,
      prefix: prefix,
      year:   year,
      number: number
    )
  end
end

def self.split(id)
  prefix, rest = id.split('-',2)
  year, number = case prefix
                 when 'CVE', 'MS'
                   # ABC-YYYY-NNNN format
                   rest.split('-',2)
                 else
                   # ABC-NNNN format
                   [nil, number]
                 end

  return prefix, year, number
end
