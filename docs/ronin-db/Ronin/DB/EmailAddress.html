def self.extract(text)
  return enum_for(__method__,text).to_a unless block_given?

  scanner = StringScanner.new(text)

  while scanner.skip_until(Support::Text::Patterns::EMAIL_ADDR)
    yield parse(scanner.matched)
  end

  return nil
end

def self.from(email)
  email = case email
          when URI::MailTo then email.to
          else                  email.to_s
          end

  return parse(email)
end

def self.parse(email)
  user, host = email.split('@',2)

  user.strip!

  if user.empty?
    raise("email address #{email.dump} must have a user name")
  end

  host.strip!

  if host.empty?
    raise("email address #{email.dump} must have a host name")
  end

  return find_or_initialize_by(
    user_name: UserName.find_or_initialize_by(name: user),
    host_name: HostName.find_or_initialize_by(address: host)
  )
end
