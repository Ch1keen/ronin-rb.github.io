def self.extract(text,version=nil)
  return enum_for(__method__,text,version).to_a unless block_given?

  IPAddr.extract(text,version) do |ip|
    yield parse(ip)
  end

  return nil
end

def self.lookup(name,nameservers=[])
  host = HostName.find_or_initialize_by(address: name)
  resolver = Network::DNS.resolver(nameservers)

  ips = begin
          resolver.getaddresses(name)
        rescue
          []
        end
    
  ips.map! do |addr|
    IPAddress.find_or_create_by(address: addr, host_names: [host])
  end

  return ips
end

def lookup!(nameservers=[])
  resolver = Network::DNS.resolver(nameservers)
  hosts = begin
            resolver.getnames(self.address.to_s)
          rescue
            []
          end

  hosts.map! do |name|
    HostName.find_or_create_by(address: name, ip_addresses: [self])
  end

  return hosts
end
