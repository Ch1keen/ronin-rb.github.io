def self.domain(name)
  address_column = self.arel_table[:address]

  name = sanitize_sql_like(name)

  where(address_column.matches("#{name}.%")).or(
    where(address_column.matches("%.#{name}.%"))
  )
end

def self.extract(text)
  return enum_for(__method__,text).to_a unless block_given?

  scanner = StringScanner.new(text)

  while scanner.skip_until(Regexp::HOST_NAME)
    yield parse(scanner.matched)
  end

  return nil
end

def self.lookup(addr,nameservers=[])
  addr = addr.to_s
  ip = IPAddress.find_or_initialize_by(address: addr)

  resolver = Network::DNS.resolver(nameservers)
  hosts = begin
            resolver.getnames(addr)
          rescue
            []
          end

  hosts.map! do |name|
    HostName.find_or_create_by(address: name, ip_addresses: [ip])
  end

  return hosts
end

def lookup!(nameservers=[])
  resolver = Network::DNS.resolver(nameservers)
  ips = begin
          resolver.getaddresses(self.address)
        rescue
          []
        end

  ips.map! do |addr|
    IPAddress.find_or_create_by(address: addr, host_names: [self])
  end

  return ips
end
