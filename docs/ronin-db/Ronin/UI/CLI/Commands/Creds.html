def print_resource(cred)
  case cred
  when ServiceCredential
    puts "#{cred}\t(#{cred.open_port.ip_address} #{cred.open_port})"
  when WebCredential
    puts "#{cred}\t(#{cred.url})"
  else
    super(cred)
  end
end
