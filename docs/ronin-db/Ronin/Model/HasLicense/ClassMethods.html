def licensed_under(license)
  conditions = case license
               when License
                 {license: license}
               when Symbol
                 unless License.methods(false).include?(license)
                   raise(ArgumentError,"unknown license: #{license}")
                 end

                 {license: License.send(license)}
               else
                 {'license.name' => license.to_s}
               end

  all(conditions)
end
