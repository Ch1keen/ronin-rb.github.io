def self.each
  ASCII.each do |c|
    yield c
    yield "#{c}#{c}" # x2
    yield "#{c}#{c}#{c}#{c}" # x4
    yield "#{c}#{c}#{c}#{c}#{c}#{c}#{c}#{c}"# x8
  end
end
