def self.each(&block)
  LONG_STRINGS.each do |string|
    LONG_LENGTHS.each do |length|
      yield string[0,length]
    end
  end

  BAD_STRINGS.each(&block)

  NULL_STRINGS.each do |string|
    SHORT_LENGTHS.each do |length|
      yield string[0,length]
    end
  end

  SHORT_BINARY_STRINGS.each(&block)

  LONG_BINARY_STRINGS.each do |string|
    SHORT_LENGTHS.each do |length|
      yield string[0,length]
    end
  end
end
