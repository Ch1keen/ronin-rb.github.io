def [](name_or_index)
  case name_or_index
  when String
    name = name_or_index

    @encoders.find { |encoder| encoder.module_name == name }
  else
    index = name_or_index

    @encoders[index]
  end
end

def encode(payload)
  encoded_payload = payload

  @encoders.each do |encoder|
    unless (encoded_payload = encoder.encode(encoded_payload))
      raise(PayloadNotEncoded,"no result was returned by the encoder: #{encoder.inspect}")
    end
  end

  return encoded_payload
end
