def campaign(name)
  condition do
    Target.first(
      'campaign.name'   => name,
      'address.address' => request.ip
    )
  end
end

def host(pattern)
  case host
  when Regexp
    condition { request.host =~ pattern }
  else
    pattern = pattern.to_s

    condition { request.host == pattern }
  end
end

def ip_address(ip)
  ip = IPAddr.new(ip.to_s) unless ip.kind_of?(IPAddr)

  condition { ip.include?(request.ip) }
end
