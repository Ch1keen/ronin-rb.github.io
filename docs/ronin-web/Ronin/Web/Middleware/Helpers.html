def response(body=[],headers={},status=200)
  response = Response.new(body,status,headers)

  yield(response) if block_given?
  return response
end

def response_for(path,headers={},status=200)
  response(
    File.new(path),
    headers.merge('Content-Type' => mime_type_for(path)),
    status
  )
end
